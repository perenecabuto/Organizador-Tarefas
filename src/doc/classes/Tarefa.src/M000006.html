    def quantidade_dias_executados
        total = 0

        total =+ 1 if self.domingo
        total =+ 1 if self.segunda
        total =+ 1 if self.terca
        total =+ 1 if self.quarta
        total =+ 1 if self.quinta
        total =+ 1 if self.sexta
        total =+ 1 if self.sabado

        return total
    end
